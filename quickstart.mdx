<Steps>
  <Step title="Get a key">
    DM [@alrerat](https://t.me/alrerat) on Telegram. You'll get a key like `pc_Kr_Ohz...`.
  </Step>

  <Step title="Check your key">
    ```bash
    curl -H "Authorization: Bearer YOUR_KEY" \
      https://polychadsbot.xyz/api/v1/me
    ```

    You'll see your tier, rate limit, and today's usage.
  </Step>

  <Step title="Pull latest alerts">
    ```bash
    curl -H "Authorization: Bearer YOUR_KEY" \
      "https://polychadsbot.xyz/api/v1/alerts/latest?limit=5"
    ```

    Returns the 5 most recent insider alerts with signal scores, prices, and wallet data.
  </Step>

  <Step title="Go real-time (Pro+)">
    ```javascript
    const ws = new WebSocket(
      "wss://polychadsbot.xyz/ws/alerts?api_key=YOUR_KEY"
    );

    ws.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      if (msg.type === "alert") {
        console.log(msg.data.event_title);
        console.log(`${msg.data.side} ${msg.data.outcome} â€” $${msg.data.usdc_value}`);
      }
    };

    // Keep-alive every 25s
    setInterval(() => {
      if (ws.readyState === WebSocket.OPEN) ws.send("ping");
    }, 25000);
    ```
  </Step>
</Steps>
